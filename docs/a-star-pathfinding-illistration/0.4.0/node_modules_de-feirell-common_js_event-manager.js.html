    (function () {
        /**
         * `EventManager` is not a real class, rather an interface which is applied with [Object.makeEventManager()]{@link makeEventManager} on an already existing Object.
         * 
         * Please see the documentation of {@link makeEventManager} for more infomration of the usage.
         * 
         * @class EventManager
         */

        /**
         * @method EventManager#addEventListener
         * @param {string} name the event which should be the trigger
         * @param {Function} func the callback function
         */
        let addEventListener = function addEventListener(events, name, func) {
            if (getTypeOf(name) != "String" || getTypeOf(func) != "Function")
                return false;

            if (!this.events.contains(name))
                throw 'the eventname "' + name + '" is not on of the definedEventNames';

            if (!(name in events))
                events[name] = [];
            else
                for (var i = 0; i < events[name].length; i++)
                    if (events[name][i] == func)
                        return false;

            events[name][events[name].length] = func;
            return true;
        }

        /**
         * @method EventManager#removeEventListener
         * @param {string} name the event on which this function was registert
         * @param {Function} func the callback function which should be removed
         */
        let removeEventListener = function removeEventListener(events, name, func) {
            if (getTypeOf(name) != "String" || getTypeOf(func) != "Function")
                return false;

            if (!this.events.contains(name))
                throw 'the eventname "' + name + '" is not on of the definedEventNames';

            let offset = 0;
            let c = [];
            if (name in events) {
                for (var i = 0; i < events[name].length - 1; i++) {
                    if (events[name][i] == func)
                        offset = 1;

                    c[i] = events[i - offset];
                }
                events[name] = c;
            }

            return offset == 1;
        }

        /**
         * @method EventManager#dispatchEvent
         * @param {string} name the event which should be triggered
         * @param {Object?} obj the event object which should be called on all registert listener,
         * if non is set then this function creates an new object, this function adds a name and
         * `timestamp` and `eventName` attribute if not present. The `timestamp` is set with `Date.now()`
         * if not present before and `eventName` is set with the given event name.
         * @returns {boolean} wether the event was dispatched or not
         */
        let dispatchEvent = function dispatchEvent(events, name, obj) {
            if (obj === null || obj === undefined)
                obj = {};

            if (getTypeOf(name) != "String" || !(obj instanceof Object))
                return false;

            if (!this.events.contains(name))
                throw 'the eventname "' + name + '" is not on of the definedEventNames: ';

            if (!(name in events))
                return false;

            if (!("timestamp" in obj))
                obj.timestamp = Date.now();

            obj.eventName = name;

            for (var i = 0; i < events[name].length; i++)
                events[name][i](obj);

            return true;
        }

        /**
         * This function transforms an Object into an {@link EventManager}.
         * 
         * This function is registert on Object as a static.
         * 
         * ```javascript
         * // SomeClass does not contains addEventListener etc.
         * let obj = new SomeClassA();
         * Object.makeEventManager(obj,['firstEvent','secondEvent']);
         * 
         * obj.addEventListener('firstEvent', () => console.log)
         * 
         * let newObj = Object.makeEventManager(,['another-event']);
         * //....
         * ```
         * 
         * ```javascript
         * // usage: you are defining some class and want this class to be able to dispatch events and provide functionality so other functions can be registert as eventlistener on this class
         * 
         * function SomeClassB(){
         *  Object.makeEventListener(this, ['someEvent', 'someOtherEvent']);
         * 
         *  // ... other initlizing code
         * 
         * }
         * 
         * SomeClassB.prototype.eventDispatchingFunction = function eventDispatchingFunction(){
         *  this.dispatchEvent('someEvent', {
         *   someAttr: 'eventAttr'
         *  })
         * }
         * 
         * // example usage
         * 
         * let obj = new SomeClassB();
         * 
         * obj.addEventListener('someEvent',event => console.log);
         * 
         * obj.eventDispatchingFunction();
         * ``` 
         * 
         * @function makeEventManager
         * @param {Object?} obj the object which should be transformed into an EventManager, if undefined it creates a new object
         * @param {string[]?} definedEvents the events this eventmanager should accept, if undefined it accepts all events
         * @returns {Object} the object which was transformed
         */
        function makeEventManager(obj, definedEvents) {
            if (obj == undefined) {
                obj = {};
            } else {
                if (!(obj instanceof Object))
                    return undefined;
            }

            let events = {};
            let definedEventNames = [];

            if (getTypeOf(definedEvents) == 'Array')
                for (var k in definedEvents)
                    if (getTypeOf(definedEvents[k]) == 'String')
                        definedEventNames[definedEventNames.length] = definedEvents[k];

            /**
             * Events contains the defined events, if non where defined the array is empty.
             * 
             * The frozen (== inmuteable) array `Events` is meant to represent the available events,
             * which includes an runtime check on register, wether or not this event is defined
             * 
             * `Events` has two special members 
             * - `toString()`: is a function which returns a string which is this array in a json format
             * - `contains()`: is a function used by add, remove and dispatch to check if the event is allowed, returns true if the array is empty
             * 
             * @member {string[]} EventManager#events
             */

            Object.defineProperties(definedEventNames, {
                toString: {
                    value: () => '[' + definedEventNames.map(v => '"' + v + '"').join(',') + ']'
                },
                contains: {
                    value: definedEventNames.length == 0 ? () => true : name => definedEventNames.indexOf(name) != -1
                }
            });

            Object.defineProperties(obj, {
                addEventListener: {
                    value: addEventListener.bind(obj, events)
                },
                removeEventListener: {
                    value: removeEventListener.bind(obj, events)
                },
                dispatchEvent: {
                    value: dispatchEvent.bind(obj, events)
                },
                events: {
                    value: Object.freeze(definedEventNames)
                }
            })

            return obj;
        }

        Object.defineProperty(Object, 'makeEventManager', {
            value: makeEventManager
        });
    })()
}
